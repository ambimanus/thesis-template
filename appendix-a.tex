\begin{algorithm}
  \caption{Random Walk Heuristik}\label{alg:rw}
  \begin{algorithmic}[1]
    \item[\textbf{Parameter:}]
    \Statex $m \defeq 10$ \Comment{Anzahl Klassen (gegeben durch Probleminstanz)}
    \Statex $l \defeq 100$ \Comment{Anzahl Wiederholungen}
    \Statex
    \item[\textbf{Funktionen:}]
    \Statex \textsc{init}() \Comment{Erzeuge initiale Lösung als \texttt{array$[1..m]$}}
    \Statex \textsc{rnd}(i) \Comment{Liefert zufälliges Element aus Klasse $i$}
    \Statex \textsc{copy}(s) \Comment{Erzeuge Kopie von $s$}
    \Statex $e^h(s)$ \Comment{Berechnet Fehler für Lösung $s$}
    \Statex
    \State $s \gets$ \textsc{init}()
    \For{$j \gets 1, l$}
      \For{$i \gets 1, m$}
        \State $c \gets$ \textsc{rnd}(i) \Comment{Wähle zufälliges Element für Klasse $i$}
        \State $s' \gets$ \textsc{copy}(s)
        \State $s'[i] \gets c$
        \If{$e^h(s') < e^h(s)$} \Comment{Vergleiche Lösungen}
          \State $s \gets s'$ \Comment{Falls besser, ersetze Lösung}
        \EndIf
      \EndFor
    \EndFor
    \State \Return $s$ \Comment{Ergebnis: beste gefundene Lösung}
  \end{algorithmic}
\end{algorithm}
